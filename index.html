document.addEventListener('DOMContentLoaded', function() { const searchBox = document.getElementById('search-box'); const searchButton = document.getElementById('search-button'); function performSearch() { const query = searchBox.value; if (query.trim() !== "") { const bingSearchUrl = `https://www.bing.com/search?q=${encodeURIComponent(query)}`; window.location.href = bingSearchUrl; } else { alert("Please enter a search term."); } } searchButton.addEventListener('click', performSearch); searchBox.addEventListener('keydown', function(event) { if (event.key === "Enter") { performSearch(); } }); });
